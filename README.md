```
class CuentaBancaria:
    def __init__(self, saldo_inicial):
        self.saldo = saldo_inicial
        self.presupuesto = {}
        self.transacciones = []

    def ingresar_dinero(self, cantidad):
        self.saldo += cantidad
        self.transacciones.append(f"Ingreso de {cantidad}")

    def gastar_dinero(self, cantidad, categoria):
        if self.saldo >= cantidad:
            self.saldo -= cantidad
            self.transacciones.append(f"Gasto de {cantidad} en {categoria}")
            if categoria in self.presupuesto:
                self.presupuesto[categoria] += cantidad
            else:
                self.presupuesto[categoria] = cantidad
        else:
            print("No tienes suficiente dinero para realizar este gasto")

    def mostrar_saldo(self):
        print(f"Tu saldo actual es: {self.saldo}")

    def mostrar_presupuesto(self):
        print("Tu presupuesto es:")
        for categoria, cantidad in self.presupuesto.items():
            print(f"{categoria}: {cantidad}")

    def mostrar_transacciones(self):
        print("Tus transacciones son:")
        for transaccion in self.transacciones:
            print(transaccion)
```
